const makeThumbnail = (dataUrl) =>
  new Promise((resolve) => {
    const img = new Image();
    img.onload = () => {
      const [mW, mH] = [160, 110];
      const r = Math.min(mW / img.width, mH / img.height, 1);
      const c = document.createElement("canvas");
      c.width  = Math.max(1, Math.round(img.width  * r));
      c.height = Math.max(1, Math.round(img.height * r));
      c.getContext("2d").drawImage(img, 0, 0, c.width, c.height);
      try { resolve(c.toDataURL("image/jpeg", 0.55)); } catch { resolve(null); }
    };
    img.onerror = () => resolve(null);
    img.src = dataUrl;
  });

// BUG FIX #2: добавлена правильная обработка ошибок сети + retry логика
const callClaude = async ({ messages, maxTokens = 1000, useSearch = false, retries = 1 }) => {
  for (let attempt = 0; attempt <= retries; attempt++) {
    try {
      const body = { model: "claude-sonnet-4-6", max_tokens: maxTokens, messages };
      if (useSearch) body.tools = [{ type: "web_search_20250305", name: "web_search" }];
      const resp = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
      });
      if (!resp.ok) {
        const e = await resp.json().catch(() => ({}));
        throw new Error(e?.error?.message || `Ошибка ${resp.status}`);
      }
      const data = await resp.json();
      if (data.error) throw new Error(data.error.message);
      // BUG FIX #3: правильная сборка текста из всех text-блоков (включая после tool_use)
      return data.content.filter(c => c.type === "text").map(c => c.text).join("\n");
    } catch (err) {
      if (attempt === retries) throw err;
      await new Promise(r => setTimeout(r, 800 * (attempt + 1))); // backoff
    }
  }
};

// BUG FIX #4: безопасный парсинг JSON с fallback
const safeParseJSON = (text) => {
  if (!text) throw new Error("Пустой ответ");
  const s = text.indexOf("{");
  const e = text.lastIndexOf("}");
  if (s === -1 || e === -1) throw new Error("JSON не найден в ответе");
  try {
    return JSON.parse(text.slice(s, e + 1));
  } catch {
    // попытка исправить обрезанный JSON
    const partial = text.slice(s);
    const fixed = partial.replace(/,\s*$/, "") + "}";
    return JSON.parse(fixed);
  }
};

                <div style={{ fontSize: "12px", marginTop: "8px", color: C.mutedLt }}>YouTube · TikTok · Telegram · Reddit · clash.gg</div>
              </div>
            )}

            {tError && (
              <div style={{ background: "#1a0808", borderRadius: "13px", padding: "16px", color: C.red, fontSize: "13px", lineHeight: 1.6 }}>
                {tError}
                <button onClick={loadTrends} style={{
                  display: "block", marginTop: "12px", background: "none",
                  border: `1px solid ${C.red}`, color: C.red, padding: "7px 14px",
                  borderRadius: "8px", cursor: "pointer", fontSize: "13px",
                }}>↺ Попробовать снова</button>
              </div>
            )}

            {trends && (
              <>
                <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
                  <div>
                    <span style={{ color: C.muted, fontSize: "12px" }}>🔥 Топ стратегий · {trends.updated}</span>
                    {trends.sources && (
                      <div style={{ marginTop: "3px", display: "flex", gap: "6px", flexWrap: "wrap" }}>
                        {trends.sources.map((s, i) => (
                          <span key={i} style={{ background: `${C.purple}18`, color: C.purple, fontSize: "10px", padding: "1px 7px", borderRadius: "5px" }}>{s}</span>
                        ))}
                      </div>
                    )}
                  </div>
                  <button onClick={loadTrends} style={{
                    background: "none", border: `1px solid ${C.border}`, color: C.muted,
                    padding: "5px 11px", borderRadius: "7px", cursor: "pointer", fontSize: "11px",
                  }}>🔄 Обновить</button>
                </div>

                {(trends.strategies || []).map((s2, idx) => {
                  const thN = s2.th?.match(/\d+/)?.[0];
                  const thC = TH_CLR[parseInt(thN)] || C.gold;
                  return (
                    <div key={idx} style={{ background: C.card, borderRadius: "16px", border: `1px solid ${C.border}`, padding: "16px" }}>
                      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: "10px" }}>
                        <div style={{ display: "flex", alignItems: "center", gap: "8px", flex: 1, flexWrap: "wrap" }}>
                          <span style={{ background: `${C.purple}33`, color: C.purple, fontSize: "10px", fontWeight: 800, padding: "2px 8px", borderRadius: "6px" }}>#{idx + 1}</span>

                          {s2.views && <span style={{ color: C.mutedLt, fontSize: "11px" }}>👁 {s2.views}</span>}
                        </div>
                        {s2.rating && <div style={{ color: C.gold, fontWeight: 900, fontSize: "15px", flexShrink: 0, marginLeft: "10px" }}>★ {s2.rating}</div>}
                      </div>
                      <div style={{ display: "flex", flexWrap: "wrap", gap: "5px", marginBottom: "10px" }}>
                        {s2.th && <span style={{ background: `${thC}22`, color: thC, fontSize: "11px", padding: "2px 9px", borderRadius: "6px", fontWeight: 700 }}>{s2.th}</span>}
                        {(s2.tags || []).map((tag, ti) => (
                          <span key={ti} style={{ background: `${C.blue}20`, color: C.blue, fontSize: "11px", padding: "2px 9px", borderRadius: "6px" }}>{tag}</span>
                        ))}
                        {s2.source && <span style={{ background: `${C.green}18`, color: C.green, fontSize: "11px", padding: "2px 9px", borderRadius: "6px" }}>📺 {s2.source}</span>}
                      </div>
                      {s2.army && <div style={{ color: C.mutedLt, fontSize: "12px", marginBottom: "6px" }}>🪖 {s2.army}</div>}
                      {s2.spells && <div style={{ color: C.mutedLt, fontSize: "12px", marginBottom: "8px" }}>⚡ {s2.spells} · 🏰 {s2.siege || "—"}</div>}
                      {s2.tip && (
                        <div style={{ background: `${C.gold}0e`, border: `1px solid ${C.gold}28`, borderRadius: "10px", padding: "10px 13px", color: C.text, fontSize: "13px", lineHeight: 1.6, marginBottom: "10px" }}>
                          💡 {s2.tip}
                        </div>
                      )}
                      <button onClick={() => { if (thN) setTh(thN); setTroops(s2.army || ""); setTab("simulate"); showToast("🎯 Стратегия выбрана!"); }}
                        style={{ width: "100%", padding: "10px", background: `${C.gold}12`, border: `1px solid ${C.gold}33`, color: C.gold, borderRadius: "10px", cursor: "pointer", fontSize: "13px", fontWeight: 700 }}>
                        🎯 Симулировать эту стратегию
                      </button>
                    </div>
                  );
                })}
              </>
            )}
          </div>
        )}

        {/* ══════════════════════════════════════════════════
            ТАБ: ВОЙСКА (АРМИИ)
        ══════════════════════════════════════════════════ */}
        {tab === "army" && (
          <div style={{ display: "flex", flexDirection: "column", gap: "12px" }}>

              <div style={{

                      </button>

                  })}
                </div>
              </div>
            </Card>

              padding: "16px", borderRadius: "14px", border: "none", width: "100%",

              cursor: armyLoading ? "not-allowed" : "pointer",
              boxShadow: armyLoading ? "none" : `0 4px 24px ${C.green}44`,
              transition: "all .25s",
            }}>
              {armyLoading
                ? <span style={{ display: "flex", alignItems: "center", justifyContent: "center", gap: "8px" }}>

                : `🌐 НАЙТИ ТОП-4 СОСТАВА ДЛЯ TH${th}`}

            {armyError && (
              <div style={{ background: "#1a0808", borderRadius: "13px", padding: "14px", color: C.red, fontSize: "13px" }}>
                {armyError}
                <button onClick={findBestArmy} style={{
                  display: "block", marginTop: "10px", background: "none",
                  border: `1px solid ${C.red}`, color: C.red, padding: "6px 12px",
                  borderRadius: "7px", cursor: "pointer", fontSize: "12px",
                }}>↺ Повторить</button>
              </div>
            )}

            {armyPicks && (
              <div style={{ display: "flex", flexDirection: "column", gap: "10px" }}>
                <div style={{ color: C.muted, fontSize: "12px", textAlign: "center" }}>
                  🔥 Актуальные мета-составы TH{th} · Июнь 2026

                {armyPicks.map((army, i) => (
                  <div key={i} style={{
                    background: C.card, borderRadius: "16px",
                    border: `1px solid ${i === 0 ? C.gold + "66" : C.border}`,
                    padding: "16px", position: "relative",
                  }}>
                    {i === 0 && (
                      <div style={{
                        position: "absolute", top: 12, right: 12,
                        background: `${C.gold}22`, color: C.gold,
                        fontSize: "10px", fontWeight: 800, padding: "2px 8px", borderRadius: "6px",
                      }}>🏆 #1 META</div>
                    )}
                    <div style={{ display: "flex", alignItems: "center", gap: "8px", marginBottom: "10px" }}>
                      <span style={{ background: `${C.purple}33`, color: C.purple, fontSize: "11px", fontWeight: 800, padding: "2px 9px", borderRadius: "6px" }}>
                        #{army.rank || i + 1}
                      </span>
                      <span style={{ fontWeight: 900, fontSize: "15px", color: i === 0 ? C.gold : C.text }}>{army.name}</span>
                    </div>
                    <div style={{ display: "flex", gap: "8px", marginBottom: "10px", flexWrap: "wrap" }}>
                      {army.winrate && <span style={{ background: `${C.green}18`, color: C.green, fontSize: "12px", fontWeight: 700, padding: "2px 10px", borderRadius: "6px" }}>✓ {army.winrate}</span>}

                      {army.source && <span style={{ background: `${C.blue}18`, color: C.blue, fontSize: "11px", padding: "2px 9px", borderRadius: "6px" }}>📺 {army.source}</span>}
                    </div>
                    <div style={{ background: C.surface, borderRadius: "10px", padding: "10px 12px", marginBottom: "10px" }}>
                      <div style={{ color: C.muted, fontSize: "10px", marginBottom: "4px" }}>🪖 СОСТАВ</div>
                      <div style={{ color: C.text, fontSize: "13px", lineHeight: 1.6, fontWeight: 600 }}>{army.army}</div>
                    </div>
                    <div style={{ color: C.mutedLt, fontSize: "12px", marginBottom: "6px" }}>⚡ {army.spells}</div>
                    <div style={{ color: C.mutedLt, fontSize: "12px", marginBottom: army.tip ? "10px" : "12px" }}>
                      🏰 {army.siege} · 🐾 {army.pet}
                    </div>
                    {army.tip && (
                      <div style={{ background: `${C.gold}0e`, border: `1px solid ${C.gold}28`, borderRadius: "10px", padding: "10px 12px", color: C.text, fontSize: "12px", lineHeight: 1.5, marginBottom: "12px" }}>
                        💡 {army.tip}
                      </div>
                    )}
                    <div style={{ display: "flex", gap: "8px" }}>
                      <button onClick={() => { setTroops(army.army + ", " + army.spells); setTab("simulate"); showToast("🎯 Состав выбран!"); }}

                        🎯 Симулировать
                      </button>
                      <button onClick={() => { setTroops(army.army + ", " + army.spells); showToast("✓ Состав выбран!"); }}

                        ✓ Выбрать
                      </button>
                    </div>
                  </div>
                ))}
              </div>
            )}
          </div>
        )}

        {/* ══════════════════════════════════════════════════
            ТАБ: ИСТОРИЯ
        ══════════════════════════════════════════════════ */}
        {tab === "history" && (
          <div style={{ display: "flex", flexDirection: "column", gap: "10px" }}>
            {history.length === 0 ? (
              <div style={{ textAlign: "center", padding: "60px 20px", color: C.muted }}>
                <div style={{ fontSize: "56px" }}>📜</div>
                <div style={{ fontWeight: 800, color: C.text, marginTop: "14px", fontSize: "16px" }}>История пуста</div>

              <>
                <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>

                    setHistory([]);
                    try { await window.storage.delete("coc_sim_v2"); } catch { /* ok */ }

                  }} style={{
                    background: "none", border: `1px solid ${C.border}`, color: C.muted,
                    padding: "5px 12px", borderRadius: "8px", cursor: "pointer", fontSize: "12px",
                  }}>🗑 Очистить</button>
                </div>

                {/* BUG FIX #11: добавлена общая статистика */}
                {history.length > 1 && (() => {
                  const avg3 = Math.round(history.filter(e => e.best_stars === 3).length / history.length * 100);
                  const avgDest = Math.round(history.reduce((s, e) => s + (e.best_dest || 0), 0) / history.length);
                  return (
                    <Card style={{ padding: "12px 14px" }}>
                      <div style={{ color: C.muted, fontSize: "10px", letterSpacing: ".8px", marginBottom: "8px" }}>📊 МОЯ СТАТИСТИКА</div>
                      <div style={{ display: "flex", gap: "20px" }}>
                        <div>
                          <div style={{ color: C.gold, fontWeight: 900, fontSize: "20px" }}>{avg3}%</div>
                          <div style={{ color: C.muted, fontSize: "10px" }}>3★ процент</div>
                        </div>
                        <div>
                          <div style={{ color: C.green, fontWeight: 900, fontSize: "20px" }}>{avgDest}%</div>
                          <div style={{ color: C.muted, fontSize: "10px" }}>сред. разруш.</div>
                        </div>
                        <div>
                          <div style={{ color: C.text, fontWeight: 900, fontSize: "20px" }}>{history.length}</div>
                          <div style={{ color: C.muted, fontSize: "10px" }}>всего игр</div>
                        </div>
                      </div>
                    </Card>
                  );
                })()}

                {history.map(entry => {
                  const thC = TH_CLR[entry.base_th] || C.gold;
                  return (
                    <div key={entry.id} style={{
                      background: C.card, borderRadius: "14px", padding: "12px",
                      border: `1px solid ${C.border}`, display: "flex", gap: "12px", alignItems: "center",
                    }}>
                      {entry.preview && (
                        <img src={entry.preview} alt="" style={{
                          width: 72, height: 52, objectFit: "cover",
                          borderRadius: "9px", flexShrink: 0,
                        }} />
                      )}
                      <div style={{ flex: 1, minWidth: 0 }}>
                        <div style={{ fontSize: "11px", color: C.muted }}>{entry.date}</div>
                        <div style={{ fontWeight: 800, marginTop: "2px", color: thC }}>
                          🏰 vs TH{entry.base_th} · моя TH{entry.th}
                        </div>
                        <div style={{ fontSize: "12px", color: C.gold, marginTop: "2px", fontWeight: 700 }}>
                          🏆 {entry.best_name}
                        </div>
                        <div style={{ fontSize: "11px", color: C.muted, marginTop: "1px" }}>
                          {STAR_LABEL[entry.best_stars]} · {entry.best_dest}% · {entry.total} симуляций
                        </div>
                      </div>
                    </div>
                  );
                })}
              </>
            )}
          </div>
        )}

      </div>

      {/* НИЖНИЙ ТАББАР */}
      <div style={{
        position: "fixed", bottom: 0, left: 0, right: 0,
        background: C.surface, borderTop: `1px solid ${C.border}`,
        display: "flex", padding: "6px 10px 8px",
        zIndex: 100, gap: "4px",
        boxShadow: "0 -4px 20px #00000060",
        // BUG FIX #12: paddingBottom учитывает safe area на iPhone/Android
        paddingBottom: "max(8px, env(safe-area-inset-bottom))",
      }}>
        {renderTabBtn("simulate", "🎯", "Симуляция")}
        {renderTabBtn("trends",  "📡", "Тренды")}
        {renderTabBtn("army",    "🪖", "Войска")}
        {renderTabBtn("history", "📜", `История (${history.length})`)}
      </div>

      <style>{`
        @keyframes spin { from { transform: rotate(0deg) } to { transform: rotate(360deg) } }
        @keyframes pulse { 0%,100%{opacity:1} 50%{opacity:.55} }
        .spin  { display: inline-block; animation: spin 1s linear infinite; }
        .pulse { animation: pulse 1.8s ease-in-out infinite; }
        * { -webkit-tap-highlight-color: transparent; box-sizing: border-box; }
        input::placeholder, textarea::placeholder { color: #3d5070; }
        ::-webkit-scrollbar { width: 4px; }
        ::-webkit-scrollbar-thumb { background: #1e3060; border-radius: 4px; }
        button:active { opacity: 0.82; }
        /* Android: убираем синюю рамку на фокусе */
        *:focus { outline: none; }
        *:focus-visible { outline: 2px solid #f5a62344; }
      `}</style>
    </div>
  );
